📜 azure-pipelines.yml

trigger:
  branches:
    include: [main]
schedules:
  - cron: "0 9 * * *"     # Daily 9 AM UTC
    branches: [main]

pool:
  vmImage: 'ubuntu-latest'

variables:
  AZURE_STORAGE_ACCOUNT: 'docqastorage'
  INPUT_CONTAINER: 'documents-in'
  OUTPUT_CONTAINER: 'reports-out'

steps:
- task: UsePythonVersion@0
  inputs: { versionSpec: '3.11' }

- script: pip install -r requirements.txt
  displayName: 'Install dependencies'

- task: AzureCLI@2
  displayName: 'Pull documents from Blob'
  inputs:
    azureSubscription: 'AzureServiceConn'
    scriptType: 'bash'
    inlineScript: |
      az storage blob download-batch -d ./docs -s $(INPUT_CONTAINER)

- script: python qa_pipeline.py --input ./docs --output ./reports
  displayName: 'Run automated QA scoring'

- task: AzureCLI@2
  displayName: 'Upload reports'
  inputs:
    azureSubscription: 'AzureServiceConn'
    scriptType: 'bash'
    inlineScript: |
      az storage blob upload-batch -d $(OUTPUT_CONTAINER) -s ./reports

- task: PublishTestResults@2
  inputs: { testResultsFiles: 'reports/test-results.xml' }
